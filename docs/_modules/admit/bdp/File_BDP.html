[docs]    def touch(self):
        """
        Touches the associated disk file.

        The implementation uses native Python OS bindings for speed.

        Parameters
        ----------
        None

        Returns
        -------
        None
        """
        fname = self.baseDir() + self.filename
        print "File_BDP: touching", fname

        if os.path.exists(fname):
            os.utime(fname,None)
        else:
            open(fname,'a').close()
